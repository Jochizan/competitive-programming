#include <iostream>
#include <cstdio>
using namespace std;

/*
Add `int max_of_four(int a, int b, int c, int d)` here.
*/
int max_of_four(int a, int b, int c, int d){
    int mayor=0;
    int ar[] = {a, b, c, d};   
    for (int i = 0; i < 4; ++i) {
        if(ar[i]>mayor){
            mayor=ar[i];
        }
    }
    return mayor;
}

int main() {
    int a, b, c, d;
    scanf("%d %d %d %d", &a, &b, &c, &d);
    int ans = max_of_four(a, b, c, d);
    printf("%d", ans);
    
    return 0;
}
